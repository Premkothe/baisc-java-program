package prem;
class Thread01 extends Thread
{
	public void run()
	{
		for(char ch = 'A';ch<='Z';ch++) 
		{
			System.out.println(ch);
			try
			{
				Thread.sleep(2000);
			}catch(InterruptedException  e)
			{
				System.out.println(e);
			}
		}
	}
	}
public class Alphabte {
	public static void main(String []args)
	{
		Thread01 t=new Thread01();
		t.start();
	}

}
